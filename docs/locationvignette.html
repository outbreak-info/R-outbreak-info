Characteristic S-gene mutations in common lineages

Mutations in at least 75% of sequences

lineage_df <- getAllLineagesByLocation("United States")

lineage_list <- unique(lineage_df$lineage)[!(unique(lineage_df$lineage) %in% "other")]

df_ll <- lineage_list %>% map_dfr(~{getMutationsByLineage(.x, frequency=0.75) %>% mutate(lineage_name = .x)})

df_ll <- subset(df_ll, gene == "S")

p <- ggplot(df_ll, aes(mutation, lineage_name, fill = prevalence)) + geom_tile() + scale_fill_gradient(trans = 'reverse', labels = scales::percent) + theme(axis.text.x = element_text(angle = 90, vjust = 0.5, hjust=1))
show(p)
